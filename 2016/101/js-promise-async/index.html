JS - promise/async

Overview: async only change the way you write a promise function.

  1. Write a promise function
//  original syntax
const sayHelloPromise = () => {
    return new Promise((resolve, reject) => {
        resolve("Hello Promise !")
    })
}

// async syntax
const sayHelloAsync = async () => {
    return "Hello Async !"
}

Outputs

sayHelloPromise().then(d => console.log(d)) // Hello Promise !
sayHelloAsync().then(d => console.log(d)) // Hello Async !
  1. Handle promise return
const userName = "jimmy"

// original syntax
const sayHelloPromiseWithName = () => {
    return this.sayHelloPromise().then(origin => {
        return origin + " and hi " + userName;
    })
}

// async syntax
const sayHelloAsyncWithName = async () => {
    const origin = await this.sayHelloAsync();
    return origin + " and hi " + userName;
}

Outputs

sayHelloPromiseWithName().then(d => console.log(d)) // Hello Promise ! and hi jimmy
sayHelloAsyncWithName().then(d => console.log(d)) // Hello Async ! and hi jimmy
  1. Error handing
const query = () => {
    return new Promise((resolv, reject) => {
        reject("some error")
    })
}

const handleQueryAsync = async () => {
    try {
        const value = await query();
        console.log(value);
    } catch (e) {
        console.log("error", e);
    }

}

const handleQueryPromise = () => {
    query()
        .then(value => {
            console.log(value);
        })
        .catch(e => {
            console.log("error", e);
        })
}

Outputs

handleQueryAsync(); // error some error
handleQueryPromise(); // error some error
